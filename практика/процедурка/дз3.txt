#include <iostream>
#include <cmath>
using namespace  std;

int main(int argc, char *argv[]){
    float S, m;
    int n;

    cout << "Enter S, m,  n: " << endl;
    cin >> S >> m >> n;

    if (S <= 0 || m <= 0) {
        cout << "you dumb" << endl;
        return 1;
    }

    if (n < 0) {
        cout << "you dumb" << endl;
        return 1;
    }

    float r = 0;
    float step = 100;
    float prev_r = 0;

    if (m * 12 * n < S) {
        cout << "you dumb" << endl;
        return 1;
    }

    while (step != 0.00001) {
        float current_m = (S * r * pow(1 + r, n)) / (12 * (pow(1 + r, n) - 1));
        if (current_m < m) {
            prev_r = r;
            r += step;
        } else if (current_m == m) {
            cout << "p:  " << r * 100 << endl;
            return 0;
        } else {
            step /= 10;
            r = prev_r + step;
        }
    }
    cout << "you dumb" << endl;
}
